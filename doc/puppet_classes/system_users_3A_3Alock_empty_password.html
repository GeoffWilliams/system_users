class system_users::lock_empty_password($lock_method = '*') {
  $users_to_lock = dig($facts, 'user_audit', 'empty_password')
  if $users_to_lock {
    user { $users_to_lock:
      password => $lock_method,
    }
  } else {
    warning("user_audit fact missing - system_users module seems broken?")
  }
}
